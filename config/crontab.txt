# 文献追踪系统 - Cron 定时任务配置
# 使用方法：
# 1. 编辑此文件中的邮箱地址
# 2. 运行：crontab /opt/data/literature-scraper/config/crontab.txt
# 3. 查看当前 cron：crontab -l

# 环境变量
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin

# 每天早晨 8:00 发送日报
0 8 * * * cd /opt/data/literature-scraper && python3 scripts/email_sender.py --type daily >> logs/email_daily.log 2>&1

# 每周日 20:00 发送周报
0 20 * * 0 cd /opt/data/literature-scraper && python3 scripts/email_sender.py --type weekly >> logs/email_weekly.log 2>&1
